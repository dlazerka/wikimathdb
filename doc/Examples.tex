Let $\|\ \| : V \rightarrow {\mathbb R}$ such that
\begin{enumerate}
\item
$\|{\bf x}\| \geq 0, \forall {\bf x} \in V$\\
$\|{\bf x}\| = 0 \Leftrightarrow {\bf x} = {\bf 0}$
\item
$\|{\bf x}\| = |\lambda|\|{\bf x}\|,\ \forall {\bf x} \in V,\
\forall \lambda \in \mathbb{R}$
\item
$\|{\bf x} + {\bf y}\| \leq \|{\bf x}\| + \|{\bf y}\|, \forall {\bf
x}, {\bf y} \in V$
\end{enumerate}

\end{enumerate}
