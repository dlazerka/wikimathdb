\begin{enumerate}
\item
Axiom of extensionality
$$
\forall x \forall y (\forall z (z \in x \Leftrightarrow z \in y)
\Rightarrow x = y)
$$

\item
Axiom of regularity
$$
\forall x (\exists y (y \in x) \Rightarrow \exists y (y \in x \wedge
\neg \exists z (z \in y \wedge z \in x)))
$$

\end{enumerate}
